// --- FASE DE INICIO (Mano) ---
escena inicio {
    decir "Estás en el mundo. Tienes tu Mano (Fuerza 1).";
    opcion "Picar Árbol (Resistencia 1)" ir_a picar_madera_obtiene_pico; // VÁLIDO (1 >= 1)
    opcion "Terminar" ir_a fin_juego;
}

// 1. Obtener Pico Madera (Fuerza 2)
escena picar_madera_obtiene_pico {
    decir "Madera obtenida. Ahora obtienes PicoMadera.";
    obtener_item PicoMadera; // Estado: PicoMadera (Fuerza 2)
    opcion "Continuar a Nivel 2" ir_a mineria_nivel_2; 
    opcion "Volver" ir_a inicio;
}
